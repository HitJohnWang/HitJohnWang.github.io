Matlab实现关键字提取并以指定格式输出

发表于 2017-04-16 |

#问题：
提取下列文本中以开头，空格作为分隔符的关键字，并在新文本中每行以指定格式<KeyWord name=”AIRBAG_SIMPLE_PRESSURE_VOLUME” />输出。

*AIRBAG_WANG_NEFSKE_MULTIPLE_JETTING *AIRBAG_LOAD_CURVE *AIRBAG_LINEAR_FLUID *AIRBAG_HYBRID *AIRBAG_HYBRID_JETTING HYBRID_CHEMKIN
*AIRBAG_SIMPLE_PRESSURE_VOLUME_POP *AIRBAG_SIMPLE_AIRBAG_MODEL_POP *AIRBAG_ADIABATIC_GAS_MODEL_POP *AIRBAG_WANG_NEFSKE_POP *AIRBAG_WANG_NEFSKE_JETTING_POP
*AIRBAG_WANG_NEFSKE_MULTIPLE_JETTING_POP *AIRBAG_LOAD_CURVE_POP *AIRBAG_LINEAR_FLUID_POP *AIRBAG_HYBRID_POP *AIRBAG_HYBRID_JETTING HYBRID_CHEMKIN_POP
*AIRBAG_SIMPLE_PRESSURE_VOLUME_CM *AIRBAG_SIMPLE_AIRBAG_MODEL_CM *AIRBAG_ADIABATIC_GAS_MODEL_CM *AIRBAG_WANG_NEFSKE_CM *AIRBAG_WANG_NEFSKE_JETTING_CM
*AIRBAG_WANG_NEFSKE_MULTIPLE_JETTING_CM *AIRBAG_LOAD_CURVE_CM *AIRBAG_LINEAR_FLUID_CM *AIRBAG_HYBRID_CM *AIRBAG_HYBRID_JETTING HYBRID_CHEMKIN_CM
*AIRBAG_SIMPLE_PRESSURE_VOLUME_ID *AIRBAG_SIMPLE_AIRBAG_MODEL_ID *AIRBAG_ADIABATIC_GAS_MODEL_ID *AIRBAG_WANG_NEFSKE_ID *AIRBAG_WANG_NEFSKE_JETTING_ID
*AIRBAG_WANG_NEFSKE_MULTIPLE_JETTING_ID *AIRBAG_LOAD_CURVE_ID *AIRBAG_LINEAR_FLUID_ID *AIRBAG_HYBRID_ID *AIRBAG_HYBRID_JETTING HYBRID_CHEMKIN_ID
*AIRBAG_ADVANCED_ALE *AIRBAG_ALE *AIRBAG_INTERACTION *AIRBAG_REFERENCE_GEOMETRY *AIRBAG_REFERENCE_GEOMETRY_BIRTH *AIRBAG_REFERENCE_GEOMETRY_RDT *AIRBAG_REFERENCE_GEOMETRY_BIRTH_RDT
*AIRBAG_SHELL_REFERENCE_GEOMETRY

#Matlab实现：

%从文本中读取指定格式的字符串，并输出
%2017/04/16 by John Wang
clear
clc
fid = fopen('keywords.uew');    
fileID=fopen('output.txt','w');% w r w+ a
tline = fgets(fid);
Row=1;
EndRow=Inf;

while ischar(tline)
 % Mehtod 1    
     expression='\*/?.*? ';                        %以'*'开始以空格结束的关键字匹配
     match=regexp(tline,expression,'match');
     Length=length(strjoin(match));                %由于最后一个关键字不以空格结尾
     tline(1:Length)=[];                        %每行最后一各关键字单独提取
     match{end+1}=tline;
     match=strtrim(match);                        %裁切字符串的开头和尾部的空格，制表，回车符

     if ~isempty(match)
         formatSpec='<keyword name="%s">\n';    %每行指定输出格式
         fprintf(fileID,formatSpec,match{1:end});
     end 

% Mehtod 2 
   expression='\*';                            
   splitStr=regexp(tline,expression,'split');    %以'*'为标识分离关键字（不包含'*'）
   splitStr=splitStr(2:end);                    %去掉空cell
   Length=length(splitStr);    
   splitStr=strtrim(splitStr);                    %裁切字符串的开头和尾部的空格，制表，回车符

   for i=1:Length
       splitStr{i}=['*' splitStr{i}];            %关键字恢复'*'号
   end

   if ~isempty(splitStr)
       formatSpec='<KeyWord name="%s" />\n';
       fprintf(fileID,formatSpec,splitStr{1:end});
   end  

    Row=Row+1;                                    %最大行判断，跳出
    if Row>EndRow
        break
    end

    tline = fgets(fid);                            %指针转至下一行
end
fclose(fid);
fclose(fileID);
